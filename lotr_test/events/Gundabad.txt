	picture = GFX_report_event_ring

	is_triggered_only = yes

	option = { #yes
		name = gundabad.2.a
		ai_chance = {
			base = 40
			modifier = {
				add = 20
				has_opinion = {
					target = ANG
					value > 75
				}
			}
			modifier = {
				add = 10
				has_non_aggression_pact_with = ANG
			}
		}
		MOR = {
			add_to_faction = GUN
		}
	}

	option = { #no

		ai_chance = {
			base = 10

					value < 25
				}
			}
		}
		ANG = {
			create_wargoal = { target = GUN type = annex_everything }
		}
	}
}
